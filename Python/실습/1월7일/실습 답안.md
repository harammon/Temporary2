#### whileLab6

~~~python
from random import *

while True:
    num = int(input("숫자를 하나 입력하세요. : "))
    if num == 0:
        print("종료")
        break
    elif num < -10 or num > 10:
        print()
        continue
    else:
        result = 1
        if num > 0:
            for i in range(1, num+1):
                result *= i 
            print("입력값 :", num)
            print(result, '\n')
        else:
            num = abs(num)
            for i in range(1, num+1):
                result *= i 
            print("입력값(부호변경) :", num)
            print(result, '\n')
~~~

#### funcLab3

~~~python
def expr(num1, num2, oper):
    if oper == '+':
        return num1 + num2
    elif oper == '-':
        return num1 - num2
    elif oper == '*':
        return num1 * num2
    elif oper == '/':
        return num1 / num2
    else:
        return None

while True:
    a, b, c = input("숫자 두 개와 연산자를 차례로 입력하세요 : ").split()
    a = int(a)
    b = int(b)
    if expr(a, b, c) == None:
        print("수행 불가")
        print()
    else:
        print("연산결과 :", expr(a, b, c))
        print()
~~~

from random import *

def differtwovalue(a, b):
    if a >= b:
        return a - b
    else:
        return b - a

for i in range(5):
    x, y = randint(1, 30), randint(1, 30)
    result = differtwovalue(x, y)
    print(x, '와', y, '의 차는', result, '입니다.')
